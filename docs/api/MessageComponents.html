 51class CodeBlock(MessageComponent):
 52    """
 53    CodeBlock: something used to represent a code block in an embed field
 54    """
 55
 56    def __init__(self, content: str, language: str = ""):
 57        """
 58        Create a new CodeBlock
 59        :param content: the code content to display
 60        :param language: the language for syntax highlighting (e.g., 'python', 'javascript', '')
 61        """
 62        super().__init__()
 63        self.content = content
 64        self.language = language
 65        self.type = "codeblock"
 66        self.limit = 1
 67
 68    def _embed_transform(self, embed: discord.Embed) -> None:
 69        """
 70        Transform the embed to add a code block field
 71        :param embed: the embed to add the code block to
 72        :return: Nothing
 73        """
 74        # Format as a code block with backticks
 75        formatted_content = f"```{self.language}\n{self.content}\n```"
 76        # Truncate if too long (Discord field value limit is 1024 chars)
 77        if len(formatted_content) > 1024:
 78            formatted_content = formatted_content[:1021] + "```"
 79        embed.add_field(name="\u200b", value=formatted_content, inline=False)

112class DataTable(MessageComponent):
113    """
114    Creates a set of fields like this
115
116    Name:      Param1:  ...
117    Tyler       Thing   ...
118    Julian      Thing2  ...
119    """
120
121    def __init__(self, data: dict) -> None:
122        """
123        Create a new data table
124        :param data: data formatted like this:
125        {Player: {"Column name": "Column Value"}, Player2: {"Column name": "Different Column Value"}, ...}
126
127        Empty parameters are automatically filled in.
128        """
129
130        # Instantiate class data
131        super().__init__()
132        self.data = data
133        self.type = "info_rows"
134        self.limit = 1  # Only one dtaa table per embed
135
136    def _embed_transform(self, embed: discord.Embed) -> None:
137        """
138        Transform the embed to add the fields that make up a data table
139        :param embed: the embed to add the fields that make up a data table
140        :return: Nothing
141        """
142
143        # Populate data in the format {data type: {person: value, person2, value}}
144        column = {}
145        for person in self.data:
146            for data_type in self.data[person]:
147                if data_type not in column:
148                    column[data_type] = {person: str(self.data[person][data_type])}
149                column[data_type].update({person: str(self.data[person][data_type])})
150
151        # Check for empty data column and add empty values
152        for data_type in column:
153            for person in self.data:
154                if person not in column[data_type]:
155                    column[data_type].update({person: ""})
156
157        # Add fields
158        number_names = 0  # Number of Name: fields that have occurred
159        for index in range((len(column) + (len(column) + 1) // 2)):
160            if index % 3 == 0:  # Every third column should be a name column because discord
161                embed.add_field(name="Name:", value="\n".join([str(p) for p in self.data.keys()]))
162                number_names += 1
163            else:  # Subtract number of names for formatting purposes
164                embed.add_field(name=list(column.keys())[index - number_names],
165                                value="\n".join(column[list(column.keys())[index - number_names]].values()))

246class Button(MessageComponent):
247    """
248    Represents a button for callbacks into the Game class
249    """
250
251    def __init__(self, label: str | None,
252                 callback: Callable[[Player, dict], typing.Any] | Callable[[Player], typing.Any], emoji: str = None,
253                 row: int | None = None, style: ButtonStyle = ButtonStyle.gray, arguments: dict = None,
254                 require_current_turn: bool = True, disabled: bool = False) -> None:
255        """
256        Create a new Button. This represents a button for callbacks into the Game class
257        :param label: label of button
258        :param callback: callback function
259        :param emoji: Emoji to use for the button
260        :param row: which row the button is in (0-4, integer)
261        :param style: coloring of button
262        :param arguments: arguments to pass to the callback function
263        :param require_current_turn: whether the button can only be used by the player whose turn it is
264        """
265
266        # Instantiate class options
267        super().__init__()
268        self.type = "button"
269        self.limit = 25
270        self.disabled = disabled
271        if not label:  # Empty string or None
272            self.name = "​"  # Zero width space for no label
273        else:
274            self.name = label  # Just pass label as given
275
276        self.style = style
277        self.callback = callback
278        self.arguments = arguments
279        self.row = row
280
281        self.emoji = parse_discord_emoji(emoji)
282
283        self.require_current_turn = require_current_turn
284        if self.arguments is not None:  # create arguments like key=value,key2=value2
285            self.parsed_arguments = ",".join([f'{key}={value}' for key, value in self.arguments.items()])
286        else:
287            self.parsed_arguments = ""  # no arguments, no problem
288
289    def _view_transform(self, view: discord.ui.View, game_id: int) -> None:
290        """
291        Add the button to the view.
292        :param view: View to add the button to.
293        :param game_id: game ID for callback
294        :return: Nothing (view object is passed as memory location)
295        """
296
297        # Note: c/ means current turn IS required, n/ means NO
298        view.add_item(discord.ui.Button(style=self.style,
299                                        label=self.name,
300                                        emoji=self.emoji,
301                                        row=self.row,
302                                        custom_id=f"{'c' if self.require_current_turn else 'n'}/{game_id}/"
303                                                  f"{self.callback.__name__}/{self.parsed_arguments}",
304                                        disabled=self.disabled))

307class Dropdown(MessageComponent):
308    """
309    Represents a dropdown for callbacks into the Game class
310    """
311
312    def __init__(self, data: list[dict],
313                 callback: Callable[[Player, dict], typing.Any] | Callable[[Player], typing.Any],
314                 row: int | None = None,
315                 require_current_turn: bool = True, min_values: int = None, max_values: int = None,
316                 placeholder: str = None, disabled: bool = False) -> None:
317        """
318        Create a new Dropdown. This represents a dropdown for callbacks into the Game class
319        :param callback: callback function
320        :param row: which row the button is in (0-4, integer)
321        :param require_current_turn: whether the button can only be used by the player whose turn it is
322        """
323
324        # Instantiate class options
325        super().__init__()
326        self.type = "select"
327        self.limit = 25
328
329        self.callback = callback
330        self.min_values = min_values
331        self.max_values = max_values
332        self.placeholder = placeholder
333        self.row = row
334        self.disabled = disabled
335
336        self.components = []
337        for component in data:
338            if "emoji" in component:
339                emoji = parse_discord_emoji(component["emoji"])
340            else:
341                emoji = None
342            if "default" not in component:
343                default = False
344            else:
345                default = component["default"]
346            description = component.get("description")
347            self.components.append(SelectOption(label=component["label"], value=component["value"],
348                                                emoji=emoji, default=default, description=description))
349
350        self.require_current_turn = require_current_turn
351
352    def _view_transform(self, view: discord.ui.View, game_id: int) -> None:
353        """
354        Add the dropdown to the view.
355        :param view: View to add the button to.
356        :param game_id: game ID for callback
357        :return: Nothing (view object is passed as memory location)
358        """
359
360        menu = discord.ui.Select(options=self.components, min_values=self.min_values,
361                                 max_values=self.max_values, placeholder=self.placeholder, disabled=self.disabled,
362                                 custom_id=f"{'select_c' if self.require_current_turn else 'select_n'}/{game_id}/"
363                                           f"{self.callback.__name__}")
364        view.add_item(menu)
