 19class _BuildContext:
 20    def __init__(self, game_id: int | None) -> None:
 21        self.game_id = game_id
 22        self.attachments: list[discord.File] = []
 23        self._attachment_names: set[str] = set()
 24        self._counter = count(1)
 25
 26    def _unique_filename(self, prefix: str, extension: str) -> str:
 27        while True:
 28            name = f"{prefix}_{next(self._counter)}.{extension}"
 29            if name not in self._attachment_names:
 30                self._attachment_names.add(name)
 31                return name
 32
 33    def prepare_attachment(
 34        self,
 35        media: str | bytes | discord.File,
 36        *,
 37        prefix: str = "attachment",
 38        extension: str = "png",
 39    ) -> str:
 40        if isinstance(media, str):
 41            return media
 42
 43        if isinstance(media, bytes):
 44            filename = self._unique_filename(prefix, extension)
 45            file = discord.File(io.BytesIO(media), filename=filename)
 46        elif isinstance(media, discord.File):
 47            filename = getattr(media, "filename", None) or self._unique_filename(prefix, extension)
 48            media.filename = filename
 49            self._attachment_names.add(filename)
 50            file = media
 51        else:
 52            raise TypeError(f"Unsupported attachment type: {type(media)!r}")
 53
 54        self.attachments.append(file)
 55        return f"attachment://{file.filename}"

 79def _normalize_row_children(children: Sequence[Any]) -> list[Any]:
 80    normalized: list[Any] = []
 81    pending_rows: dict[int, list[Button | Select]] = {}
 82
 83    def flush_rows() -> None:
 84        if not pending_rows:
 85            return
 86        for row_index in sorted(pending_rows):
 87            normalized.append(ActionRow(*pending_rows[row_index]))
 88        pending_rows.clear()
 89
 90    for child in children:
 91        if child is None:
 92            continue
 93        if isinstance(child, (Button, Select)):
 94            row_index = child.row if child.row is not None else 0
 95            pending_rows.setdefault(row_index, []).append(child)
 96        else:
 97            flush_rows()
 98            normalized.append(child)
 99
100    flush_rows()
101    return normalized

108def format_data_table(data: dict[Any, dict[str, Any]]) -> str:
109    if not data:
110        return "_No data_"
111
112    column_names: list[str] = []
113    for values in data.values():
114        for key in values:
115            if key not in column_names:
116                column_names.append(str(key))
117
118    rows: list[list[str]] = []
119    for player, values in data.items():
120        row = [str(player)]
121        for column in column_names:
122            row.append(str(values.get(column, "")))
123        rows.append(row)
124
125    headers = ["Name", *column_names]
126    widths = [len(header) for header in headers]
127    for row in rows:
128        for index, value in enumerate(row):
129            widths[index] = max(widths[index], len(value))
130
131    def fmt_row(values: Sequence[str]) -> str:
132        return " | ".join(value.ljust(widths[index]) for index, value in enumerate(values))
133
134    sep = "-+-".join("-" * width for width in widths)
135    lines = [fmt_row(headers), sep, *(fmt_row(row) for row in rows)]
136    return code_block("\n".join(lines))
